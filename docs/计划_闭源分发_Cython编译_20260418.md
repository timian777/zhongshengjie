## §2 创建 setup.py（Cython 编译入口）

### 2.1 在项目根创建 `setup.py`，内容：

```python
"""
众生界 · Cython 编译配置
仅用于闭源发版（M6）。普通用户运行不需要。

用法：
    python setup.py build_ext --inplace

编译范围：core/ 下所有 .py 文件，但不含任何 __init__.py
产物：core/**/*.cp312-win_amd64.pyd（与原 .py 同目录）
"""

import sys
from pathlib import Path
from setuptools import setup
from Cython.Build import cythonize
from Cython.Compiler import Options

# Cython 编译选项
Options.docstrings = True   # 保留 docstring（决策 Q5=c：本地 master 调试友好）
Options.embed_pos_in_docstring = False
Options.generate_cleanup_code = False
Options.fast_fail = True    # 任一文件编译失败立即停止


def collect_core_modules():
    """收集 core/ 下所有需要编译的 .py 文件，排除 __init__.py 和测试文件"""
    core_dir = Path("core")
    if not core_dir.exists():
        sys.exit("ERROR: core/ 目录不存在，请在项目根运行")

    py_files = []
    for p in core_dir.rglob("*.py"):
        # 排除规则
        if p.name == "__init__.py":
            continue
        if p.name.startswith("test_"):
            continue
        if "__pycache__" in p.parts:
            continue
        py_files.append(str(p))
    return py_files


def main():
    modules = collect_core_modules()
    print(f"[setup.py] 即将编译 {len(modules)} 个 .py 文件 → .pyd")
    for m in modules:
        print(f"  {m}")

    setup(
        name="zhongshengjie",
        version="0.1.0",
        ext_modules=cythonize(
            modules,
            compiler_directives={
                "language_level": 3,
                "always_allow_keywords": True,    # 兼容 argparse / dict-style kwargs
                "binding": True,                   # 保留函数签名 introspection
                "embedsignature": True,            # docstring 含签名
                "boundscheck": True,               # 数组越界检查（非性能关键，安全优先）
                "wraparound": True,
            },
            nthreads=4,                            # 并行编译，4 线程
        ),
        zip_safe=False,
    )


if __name__ == "__main__":
    main()
```

## §3 创建 build.py（编译驱动脚本）

### 3.1 在项目根创建 `build.py`，内容：

```python
"""
众生界 · 编译驱动
封装 setup.py + 清理 + 校验为单一入口。

用法：
    python build.py            # 编译 + 校验
    python build.py --clean    # 仅清理（删 .c / .pyd / build/）
    python build.py --rebuild  # 先清理再编译

执行环境：master 分支（保留 .py 源），不要在 release-closed 分支跑
"""

import argparse
import shutil
import subprocess
import sys
from pathlib import Path


CORE_DIR = Path("core")
BUILD_DIR = Path("build")
PYD_PATTERN = "*.cp*-win_amd64.pyd"
C_PATTERN = "*.c"


def clean():
    """删除所有编译产物：.c 中间文件、.pyd 二进制、build/ 缓存"""
    print("[clean] 清理编译产物...")

    # 删 build/
    if BUILD_DIR.exists():
        shutil.rmtree(BUILD_DIR)
        print(f"  removed {BUILD_DIR}/")

    # 删 core/ 下所有 .c 和 .pyd
    deleted = 0
    for p in CORE_DIR.rglob(C_PATTERN):
        # 跳过非 Cython 产生的 .c（项目几乎不会有）
        if p.with_suffix(".py").exists():
            p.unlink()
            deleted += 1
    for p in CORE_DIR.rglob(PYD_PATTERN):
        p.unlink()
        deleted += 1
    print(f"  removed {deleted} 个 .c/.pyd 文件")


def compile_all():
    """调用 setup.py build_ext --inplace"""
    print("[compile] 启动 Cython 编译...")
    result = subprocess.run(
        [sys.executable, "setup.py", "build_ext", "--inplace"],
        check=False,
    )
    if result.returncode != 0:
        sys.exit(f"[compile] FAIL exit={result.returncode}")
    print("[compile] 编译完成")


def verify_pyd_exists():
    """校验每个待编译的 .py 都有对应 .pyd"""
    print("[verify] 校验 .pyd 完整性...")
    expected_py = []
    for p in CORE_DIR.rglob("*.py"):
        if p.name == "__init__.py":
            continue
        if "__pycache__" in p.parts:
            continue
        expected_py.append(p)

    missing = []
    for py in expected_py:
        # 对应 .pyd：core/foo.py → core/foo.cp312-win_amd64.pyd
        # 用 glob 匹配任何 cpXX-win_amd64.pyd
        pyd_glob = list(py.parent.glob(f"{py.stem}.cp*-win_amd64.pyd"))
        if not pyd_glob:
            missing.append(py)

    if missing:
        print(f"[verify] FAIL · {len(missing)} 个 .py 没有对应 .pyd:")
        for p in missing:
            print(f"  {p}")
        sys.exit(1)
    print(f"[verify] OK · {len(expected_py)} 个 .py 都有对应 .pyd")


def post_check_import():
    """快速 import core 验证 .pyd 真的可加载"""
    print("[post-check] 临时移走 .py 源测试 .pyd 是否被优先加载...")
    # 我们不真的移走 .py（master 要保留），改为 importlib 直接 load .pyd
    import importlib
    import core
    importlib.reload(core)
    # 试加载几个关键模块
    for mod_name in [
        "core.cli",
        "core.config_loader",
        "core.conversation.intent_classifier",
        "core.retrieval.unified_retrieval_api",
    ]:
        try:
            m = importlib.import_module(mod_name)
            # 检查模块文件后缀（应是 .pyd 或 .py，看 import 路径）
            f = getattr(m, "__file__", "?")
            print(f"  {mod_name}: loaded from {f}")
        except Exception as e:
            print(f"  {mod_name}: FAIL {e}")
            sys.exit(1)
    print("[post-check] OK")


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--clean", action="store_true", help="仅清理")
    parser.add_argument("--rebuild", action="store_true", help="清理再编译")
    args = parser.parse_args()

    if args.clean:
        clean()
        return

    if args.rebuild:
        clean()

    compile_all()
    verify_pyd_exists()
    post_check_import()
    print("\n[build.py] ALL GREEN ✅")


if __name__ == "__main__":
    main()
```

## §4 创建 release.py（master → release-closed 分支自动化）

### 4.1 在项目根创建 `release.py`，内容：

```python
"""
众生界 · 闭源发布脚本
执行流程：
    1. 校验当前在 master 分支且 tree clean
    2. 执行 build.py（编译 .pyd）
    3. 切换/创建 release-closed 分支
    4. 把 master 当前 commit 全量同步到 release-closed
    5. 在 release-closed 上删除所有 core/ 下的 .py（保留 __init__.py）
    6. .gitignore 切换为 release 模式
    7. git add + 显示 status，但不 commit、不 push（由用户最后拍板）

用法：
    python release.py            # 全流程，最后停在 release-closed 待 commit
    python release.py --abort    # 切回 master 并清理 .pyd
"""

import argparse
import shutil
import subprocess
import sys
from pathlib import Path


RELEASE_BRANCH = "release-closed"
GITIGNORE_RELEASE = """# ========================================
# 众生界 · release-closed 分支 .gitignore
# ========================================
# 此分支只发布编译后的 .pyd + 配置 + 资源
# 任何 core/ 下的 .py 源都不应该出现在这里
# ========================================

# core/ 源码全屏蔽（仅保留 __init__.py 和 .pyd）
core/**/*.py
!core/**/__init__.py
!core/__init__.py

# 编译中间产物
core/**/*.c
build/
*.egg-info/

# Python 字节码
__pycache__/
*.pyc

# 测试相关（release 不带测试）
tests/
pytest.ini
.pytest_cache/

# 开发文档（release 仅保留 README）
docs/
PROJECT_LOGIC.md

# 其他与 master 一致的排除（用户数据等）
创作技法/**/*.md
设定/
正文/
章节大纲/
章节经验日志/
写作标准积累/
评估报告/
/总大纲.md
.case-library/cases/
.vectorstore/
.archived/
archived/
.claude/
.obsidian/
.workbuddy/
.opencode/
.cache/
.serena/
.state/
.workflow_states/
.project_memory/
"""


def run(cmd, check=True, capture=False):
    """运行命令，可选捕获输出"""
    print(f"  $ {' '.join(cmd) if isinstance(cmd, list) else cmd}")
    if capture:
        return subprocess.run(cmd, check=check, capture_output=True, text=True, shell=isinstance(cmd, str))
    return subprocess.run(cmd, check=check, shell=isinstance(cmd, str))


def assert_master_clean():
    """断言当前在 master 且 tree clean"""
    branch = run(["git", "rev-parse", "--abbrev-ref", "HEAD"], capture=True).stdout.strip()
    if branch != "master":
        sys.exit(f"ERROR: 当前分支 {branch}，必须在 master 才能 release")
    status = run(["git", "status", "--porcelain"], capture=True).stdout.strip()
    if status:
        sys.exit(f"ERROR: master 有未 commit 改动:\n{status}\n请先 commit 或 stash")
    print("[check] master 干净")


def ensure_release_branch():
    """切换到 release-closed 分支（不存在则从 master 创建）"""
    branches = run(["git", "branch", "--list", RELEASE_BRANCH], capture=True).stdout.strip()
    if branches:
        print(f"[branch] {RELEASE_BRANCH} 已存在，先 reset 到 master")
        run(["git", "checkout", RELEASE_BRANCH])
        run(["git", "reset", "--hard", "master"])
    else:
        print(f"[branch] 创建 {RELEASE_BRANCH}（基于 master）")
        run(["git", "checkout", "-b", RELEASE_BRANCH, "master"])


def strip_py_sources():
    """删除 core/ 下除 __init__.py 外所有 .py（.pyd 保留）"""
    print("[strip] 删除 core/ 下所有 .py 源（保留 __init__.py 和 .pyd）...")
    deleted = 0
    for p in Path("core").rglob("*.py"):
        if p.name == "__init__.py":
            continue
        if "__pycache__" in p.parts:
            continue
        p.unlink()
        deleted += 1
    print(f"[strip] 删除 {deleted} 个 .py")


def write_release_gitignore():
    """覆盖 .gitignore 为 release 版本"""
    print("[gitignore] 写入 release 版本 .gitignore")
    Path(".gitignore").write_text(GITIGNORE_RELEASE, encoding="utf-8")


def stage_changes():
    """git add 全部改动，但不 commit"""
    print("[stage] git add -A（注意：用户应在 commit 前 review）")
    run(["git", "add", "-A"])
    run(["git", "status"])


def abort():
    """切回 master 并清理 .pyd"""
    branch = run(["git", "rev-parse", "--abbrev-ref", "HEAD"], capture=True).stdout.strip()
    if branch == RELEASE_BRANCH:
        run(["git", "checkout", "master"])
    run(["git", "reset", "--hard", "HEAD"], check=False)
    print("[abort] 已切回 master，清理 .pyd")
    run([sys.executable, "build.py", "--clean"])


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--abort", action="store_true", help="切回 master 并清理")
    args = parser.parse_args()

    if args.abort:
        abort()
        return

    print("=" * 60)
    print("众生界 · 闭源发布流程")
    print("=" * 60)

    # 1. 校验
    assert_master_clean()

    # 2. 编译
    print("\n[1/5] 编译 .pyd...")
    run([sys.executable, "build.py"])

    # 3. 分支切换
    print(f"\n[2/5] 切到 {RELEASE_BRANCH} 分支...")
    ensure_release_branch()

    # 4. 删 .py 源
    print("\n[3/5] 剥离 .py 源...")
    strip_py_sources()

    # 5. 替换 .gitignore
    print("\n[4/5] 切换 .gitignore...")
    write_release_gitignore()

    # 6. stage（不 commit）
    print("\n[5/5] git add（不 commit）...")
    stage_changes()

    print("\n" + "=" * 60)
    print(f"✅ release-closed 分支已就绪")
    print("=" * 60)
    print(f"""
下一步（由用户手动）：
  1. git diff --cached  # review 改动
  2. git commit -m "release: v0.1.0 闭源编译版本"
  3. git push origin {RELEASE_BRANCH}
  4. 测试 clean clone:
     git clone -b {RELEASE_BRANCH} <repo-url> /tmp/test-clone
     cd /tmp/test-clone && python -m core --help

切回 master 用：
  python release.py --abort
""")


if __name__ == "__main__":
    main()
```
