## Task 3：创建 tools/init_novel.py（新用户引导脚本）

**文件：**

**背景：** 学生克隆项目后，需要：(1) 创建自己的 world config JSON，(2) 更新 config.json 中的 worldview.current_world。`init_novel.py` 自动完成这两步，并打印下一步指引。

- [ ] **Step 1: 创建 tools/init_novel.py**

```python
# tools/init_novel.py
"""新小说项目初始化工具。

用法:
    python tools/init_novel.py --name "我的小说"
    python tools/init_novel.py --name "星海纪元" --template 科幻世界示例
"""
import argparse
import json
import shutil
import sys
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent
WORLDS_DIR = PROJECT_ROOT / "config" / "worlds"
CONFIG_PATH = PROJECT_ROOT / "config.json"


def list_templates() -> list[str]:
    return [p.stem for p in WORLDS_DIR.glob("*.json") if "示例" in p.stem or p.stem == "修仙世界示例"]


def create_world_config(novel_name: str, template: str) -> Path:
    template_path = WORLDS_DIR / f"{template}.json"
    target_path = WORLDS_DIR / f"{novel_name}.json"

    if not template_path.exists():
        print(f"[错误] 模板文件不存在: {template_path}")
        sys.exit(1)

    if target_path.exists():
        print(f"[跳过] 世界配置已存在: {target_path}")
        return target_path

    data = json.loads(template_path.read_text(encoding="utf-8"))
    data["world_name"] = novel_name
    data["description"] = f"{novel_name} — 请在此填写世界观描述"

    target_path.write_text(
        json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8"
    )
    print(f"[创建] 世界配置: {target_path}")
    return target_path


def update_config_json(novel_name: str) -> None:
    if not CONFIG_PATH.exists():
        print(f"[警告] config.json 不存在，请先复制 config.example.json")
        return

    data = json.loads(CONFIG_PATH.read_text(encoding="utf-8"))
    old_world = data.get("worldview", {}).get("current_world", "未设置")

    if "worldview" not in data:
        data["worldview"] = {}
    data["worldview"]["current_world"] = novel_name

    CONFIG_PATH.write_text(
        json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8"
    )
    print(f"[更新] config.json worldview.current_world: {old_world!r} → {novel_name!r}")


def main():
    parser = argparse.ArgumentParser(description="初始化新小说项目")
    parser.add_argument("--name", required=True, help="小说名称，如：我的玄幻世界")
    parser.add_argument(
        "--template",
        default="修仙世界示例",
        help="使用哪个模板（默认：修仙世界示例）",
    )
    parser.add_argument(
        "--list-templates",
        action="store_true",
        help="列出可用模板",
    )
    args = parser.parse_args()

    if args.list_templates:
        templates = list_templates()
        print("可用模板：")
        for t in templates:
            print(f"  {t}")
        return

    novel_name = args.name.strip()
    if not novel_name:
        print("[错误] 小说名称不能为空")
        sys.exit(1)

    print(f"\n=== 初始化小说项目：{novel_name} ===\n")

    world_path = create_world_config(novel_name, args.template)
    update_config_json(novel_name)

    print(f"""
=== 初始化完成 ===

下一步：
1. 编辑世界观配置：
   {world_path}

2. 填写以下字段（基于你的小说设定）：
   - power_systems   力量体系（修仙/魔法/科技等）
   - factions        主要势力
   - key_characters  核心角色
   - core_principles 世界核心原则

3. 设置 Qdrant 向量库（参考 README.md § 快速开始）

4. 开始创作：
   python -m core write --chapter 第一章
""")


if __name__ == "__main__":
    main()
```

# tests/test_multi_novel_config.py
"""验证多小说配置切换逻辑（P5 多小说解耦）。"""
import json
import sys
from pathlib import Path
import pytest

PROJECT_ROOT = Path(__file__).resolve().parent.parent
sys.path.insert(0, str(PROJECT_ROOT))


def test_get_current_world_returns_string():
    """get_current_world() 返回非空字符串"""
    from core.config_loader import get_current_world
    world = get_current_world()
    assert isinstance(world, str)
    assert len(world) > 0


def test_world_config_file_exists_for_current_world():
    """当前世界观对应的 config/worlds/*.json 文件存在"""
    from core.config_loader import get_current_world
    world = get_current_world()
    world_config_path = PROJECT_ROOT / "config" / "worlds" / f"{world}.json"
    assert world_config_path.exists(), (
        f"config/worlds/{world}.json 不存在，请先创建或检查 config.json worldview.current_world"
    )


def test_world_config_has_required_fields():
    """world config JSON 包含必要字段"""
    from core.config_loader import get_current_world
    world = get_current_world()
    path = PROJECT_ROOT / "config" / "worlds" / f"{world}.json"
    if not path.exists():
        pytest.skip(f"world config {world}.json not found")
    data = json.loads(path.read_text(encoding="utf-8"))
    for field in ["world_name", "world_type", "power_systems", "factions", "core_principles"]:
        assert field in data, f"world config 缺少字段: {field}"


def test_example_world_configs_are_valid_json():
    """config/worlds/ 下所有 JSON 文件均可正常解析"""
    worlds_dir = PROJECT_ROOT / "config" / "worlds"
    json_files = list(worlds_dir.glob("*.json"))
    assert len(json_files) > 0, "config/worlds/ 目录为空"
    for f in json_files:
        data = json.loads(f.read_text(encoding="utf-8"))
        assert "world_name" in data, f"{f.name} 缺少 world_name 字段"


def test_init_novel_creates_world_config(tmp_path):
    """init_novel.py 创建的 world config 结构正确"""
    import importlib.util, sys as _sys
    spec = importlib.util.spec_from_file_location(
        "init_novel", PROJECT_ROOT / "tools" / "init_novel.py"
    )
    mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(mod)

    # 临时使用 tmp_path 作为 worlds dir
    import unittest.mock as mock
    with mock.patch.object(mod, "WORLDS_DIR", PROJECT_ROOT / "config" / "worlds"):
        with mock.patch.object(mod, "CONFIG_PATH", tmp_path / "config.json"):
            (tmp_path / "config.json").write_text(
                json.dumps({"worldview": {"current_world": "旧小说"}}), encoding="utf-8"
            )
            target = mod.create_world_config("新小说测试", "修仙世界示例")
            mod.update_config_json("新小说测试")

    assert target.exists()
    data = json.loads(target.read_text(encoding="utf-8"))
    assert data["world_name"] == "新小说测试"

    cfg = json.loads((tmp_path / "config.json").read_text(encoding="utf-8"))
    assert cfg["worldview"]["current_world"] == "新小说测试"

    # 清理
    target.unlink()
